6pub fn dataset_fft(dataset: &Dataset, parallel: bool) -> Dataset {
7    let freq_features: Vec<Vec<f64>> = if parallel {
8        dataset
9            .features
10            .par_iter()
11            .map(|sample| {
12                let len = sample.len();
13                let mut planner = FftPlanner::new();
14                let fft = planner.plan_fft_forward(len);
15                let mut buffer: Vec<Complex<f64>> =
16                    sample.iter().map(|&x| Complex { re: x, im: 0.0 }).collect();
17                fft.process(&mut buffer);
18                let mut spectrum = Vec::with_capacity(2 * len);
19                for c in buffer {
20                    spectrum.push(c.re);
21                    spectrum.push(c.im);
22                }
23                spectrum
24            })
25            .collect()
26    } else {
27        dataset
28            .features
29            .iter()
30            .map(|sample| {
31                let len = sample.len();
32                let mut planner = FftPlanner::new();
33                let fft = planner.plan_fft_forward(len);
34                let mut buffer: Vec<Complex<f64>> =
35                    sample.iter().map(|&x| Complex { re: x, im: 0.0 }).collect();
36                fft.process(&mut buffer);
37                let mut spectrum = Vec::with_capacity(2 * len);
38                for c in buffer {
39                    spectrum.push(c.re);
40                    spectrum.push(c.im);
41                }
42                spectrum
43            })
44            .collect()
45    };
46
47    Dataset {
48        features: freq_features,
49        labels: dataset.labels.clone(),
50    }
51}

53/// Reconstruct from [re0, im0, re1, im1, ...] using inverse FFT.
54pub fn dataset_ifft(dataset: &Dataset, parallel: bool) -> Dataset {
55    let time_features: Vec<Vec<f64>> = if parallel {
56        dataset
57            .features
58            .par_iter()
59            .map(|sample| {
60                let len = sample.len() / 2;
61                let mut planner = FftPlanner::new();
62                let ifft = planner.plan_fft_inverse(len);
63                let mut buffer: Vec<Complex<f64>> = (0..len)
64                    .map(|i| Complex {
65                        re: sample[2 * i],
66                        im: sample[2 * i + 1],
67                    })
68                    .collect();
69                ifft.process(&mut buffer);
70                buffer.iter().map(|c| c.re / len as f64).collect()
71            })
72            .collect()
73    } else {
74        dataset
75            .features
76            .iter()
77            .map(|sample| {
78                let len = sample.len() / 2;
79                let mut planner = FftPlanner::new();
80                let ifft = planner.plan_fft_inverse(len);
81                let mut buffer: Vec<Complex<f64>> = (0..len)
82                    .map(|i| Complex {
83                        re: sample[2 * i],
84                        im: sample[2 * i + 1],
85                    })
86                    .collect();
87                ifft.process(&mut buffer);
88                buffer.iter().map(|c| c.re / len as f64).collect()
89            })
90            .collect()
91    };
92
93    Dataset {
94        features: time_features,
95        labels: dataset.labels.clone(),
96    }
97}

99/// maximum absolute difference between two Datasets and check if all differences are within a tolerance.
100pub fn compare_datasets_within_tolerance(
101    original: &Dataset,
102    reconstructed: &Dataset,
103    tolerance: f64,
104) -> (f64, bool) {
105    let mut max_diff = 0.0;
106    let mut all_within = true;
107
108    for (orig_sample, recon_sample) in original.features.iter().zip(&reconstructed.features) {
109        for (&orig, &recon) in orig_sample.iter().zip(recon_sample.iter()) {
110            let diff = (orig - recon).abs();
111            if diff > max_diff {
112                max_diff = diff;
113            }
114            if diff > tolerance {
115                all_within = false;
116            }
117        }
118    }
119
120    (max_diff, all_within)
121}
